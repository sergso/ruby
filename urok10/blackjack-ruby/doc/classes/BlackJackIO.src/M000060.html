  def get_num_decks
    result = nil
    while not result
      print "Enter number of decks [#{Shoe::MIN_DECKS}.."
      print "#{Shoe::MAX_DECKS}] [1]: "
      result = gets
      result = result.strip
      if result.to_i >= Shoe::MIN_DECKS and result.to_i <= Shoe::MAX_DECKS
        return result.to_i
      elsif result.downcase == "q"
        throw :quit
      elsif result.downcase == "i"
        result = nil
        instructions
      elsif result == ""
        return Shoe::MIN_DECKS
      else
        result = nil
        try_again
      end
    end
  end
