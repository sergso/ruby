  def change_bet_sizes
    result = nil
    while not result
      print "Place your bets.  Change bets [y|n] [n]? "
      result = gets
      result = result.strip.downcase
      case result
      when "y"
        return 1
      when "n", ""
        return nil
      when "q"
        throw :quit
      when "i"
        instructions
      end
      result = nil
      try_again
    end
  end
