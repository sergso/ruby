  def place_your_bets
    # Check to see if we need to ask for bet sizes
    if @num_rounds == 0 or need_bet_change or @io.change_bet_sizes
      @io.get_players_bet_sizes(@players, @min_bet, @max_bet, @bet_increment)
    end

    for player in @players
      player.start_hand
    end
  end
