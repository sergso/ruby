  def hand_to_s(hand)
    result = @name
    i = @hands.index(hand)
    if i and @hands.length > 1 and not self.is_dealer
      result += " [Hand #{i + 1}] [\$#{hand.bet.to_s}]"
    elsif self.has_a_hand and not self.is_dealer
      result += " [\$#{@bet.to_s}]"
    end
    result += " [\$#{@bankroll.to_s}]"
    return result
  end
