  def split_hand(hand)
    if self.can_split_hand(hand)
      @num_hands += 1
      @@total_hands += 1
      i = @hands.length
      @hands[i] = hand.split
      self.place_bet(@hands[i])
      return @hands[i]
    else
      raise ArgumentError, "Hand cannot be split: #{hand.to_s}"
    end
  end
