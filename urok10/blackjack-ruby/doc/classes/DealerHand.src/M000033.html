  def total
    tot = 0
    for card in @cards
      tot += card.value
    end
    if has_ace and tot < 12
      tot2 = tot + 10
      if tot2 > 17
        return tot2
      elsif tot2 == 17
        unless @hit_soft_17
          return tot2
        end
      end
    end
    return tot
  end
