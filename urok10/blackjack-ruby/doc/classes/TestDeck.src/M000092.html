  def test_deck_cut
    1.upto(3) do |x|
      cut_at = rand(51) + 1
      x = @cards[cut_at - 1]
      y = @cards[cut_at]
      @deck.cut(cut_at)
      print "\nCut Deck at #{cut_at} [#{x}|#{y}]:\n#{@deck.to_s}\n\n"
      assert_equal(52, @cards.length, "Num Cards")
      assert_equal(@cards[(@cards.length - 1)], x, "Card Before")
      assert_equal(@cards[0], y, "Card After")
    end
  end
